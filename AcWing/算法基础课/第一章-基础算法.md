    int n,s,t,cnt=0,pre_end=-inf;
    scanf("%d",&n);
    for(int i=0;i<n;i++){
        scanf("%d %d",&s,&t);
        A[i]={s,t};
    }
    sort(A,A+n);
    for(int i=0;i<n;i++){
        if(A[i].first>pre_end)cnt++;
        pre_end=Max(pre_end,A[i].second);
    }
    printf("%d",cnt);
    return 0;
}
